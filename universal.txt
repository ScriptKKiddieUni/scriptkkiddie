-- Load OrionLib
local OrionLib = loadstring(game:HttpGet(('https://raw.githubusercontent.com/shlexware/Orion/main/source')))()

-- Create a window with a beautiful theme
local Window = OrionLib:MakeWindow({
    Name = "ScriptKKiddie Universal 🌌", -- Window name with an emoji
    HidePremium = false,
    SaveConfig = true,
    ConfigFolder = "YourConfig",
    IntroText = "Welcome to ScriptKKiddie! 🎉",
    IntroDuration = 3,
})

-- Function to copy link to clipboard
local function copyToClipboard(link)
    setclipboard(link) -- Copies the provided link to the clipboard
end

-- Get the player's username
local player = game.Players.LocalPlayer
local username = player.Name

-- Add the Universal tab with an emoji
local UniversalTab = Window:MakeTab({
    Name = "Universal ⚔️", -- Tab name with an emoji
    Icon = "rbxassetid://4483345998", -- Replace with any custom icon
    PremiumOnly = false
})

-- Add buttons for each game in the Universal tab
local games = {
    "Murder Mystery 2 🔪",
    "Adopt Me 🐾",
    "Blade Ball 🏐",
    "Da Hood 🏙️",
    "Bloxfruits 🍉",
    "Toilet Tower Defense 🚽",
    "Anime Defenders 🛡️",
    "Jailbreak 🚔",
    "Murder vs Sheriff 🚓",
    "Auto Build (BABFT) 🏗️", -- Added Auto Build to the list
    "Build a Boat For Treasure 🛥️",
    "Pls Donate 💵",
    "Legend Of Speed ⚡",
    "Brookhaven 🏡",
    "Pets Go 🐾" -- New game button for Pets Go
}

for _, gameName in ipairs(games) do
    UniversalTab:AddButton({
        Name = gameName, -- Button name
        Callback = function()
            -- Specific functionality for Murder Mystery 2
            if gameName == "Murder Mystery 2 🔪" then
                loadstring(game:HttpGet("https://raw.githubusercontent.com/ScriptKKiddieUni/mm2halloween/refs/heads/main/support%20mid%20level%20exec.lua", true))()
                print("Murder Mystery 2 script executed!") -- Log action
                return
            end
            
            -- Specific functionality for Auto Build (BABFT)
            if gameName == "Auto Build (BABFT) 🏗️" then
                loadstring(game:HttpGet("https://raw.githubusercontent.com/ScriptKKiddieUni/build-a-boat-treasure/refs/heads/main/autobuild", true))()
                print("Auto Build script executed!") -- Log action
                return
            end

            -- Specific functionality for Pets Go
            if gameName == "Pets Go 🐾" then
                loadstring(game:HttpGet("https://raw.githubusercontent.com/ScriptKKiddieUni/petsgosolara/refs/heads/main/petsgo.lua", true))()
                print("Pets Go script executed!") -- Log action
                return
            end

            -- Specific functionality for Bloxfruits
            if gameName == "Bloxfruits 🍉" then
                loadstring(game:HttpGet("https://raw.githubusercontent.com/ScriptKKiddieUni/bloxfruits/main/script.lua", true))()
                print("Bloxfruits script executed!") -- Log action
                return
            end

            -- Copy the Discord link to the clipboard
            copyToClipboard("https://discord.gg/RYdnzPudJd")

            -- Show notifications when the button is clicked
            game.StarterGui:SetCore("SendNotification", {
                Title = "ScriptKKiddie",
                Text = "Thank you for using my script! 🎉",
                Duration = 3 -- Duration for the first notification
            })

            wait(3) -- Wait for the first notification to finish before showing the next

            game.StarterGui:SetCore("SendNotification", {
                Title = "Discord Link",
                Text = "Discord link copied to your clipboard! 📋",
                Duration = 3 -- Duration for the second notification
            })

            print(gameName .. " button pressed!") -- Log which button was pressed
        end
    })
end

-- Other tabs and functionality remain the same
-- ...

-- Finalize the script
OrionLib:Init()
